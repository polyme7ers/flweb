import pandas as pd
import matplotlib.pyplot as plt
from sklearn.preprocessing import StandardScaler
from sklearn.cluster import KMeans

data = pd.read_csv("sales_data_sample.csv", encoding='ISO-8859-1')
#print(data.head())
print(data.info())

X = data[['QUANTITYORDERED', 'PRICEEACH', 'SALES']]

scaler = StandardScaler()
X_scaled = scaler.fit_transform(X)


# Step 5: Find Optimal K (Elbow Method)
inertia = []
for k in range(1, 6):
    km = KMeans(n_clusters=k, random_state=42)
    km.fit(X_scaled)
    inertia.append(km.inertia_)

plt.plot(range(1, 6), inertia, marker='o')
plt.title('Elbow Method')
plt.xlabel('Number of Clusters (k)')
plt.ylabel('Inertia')
plt.show()

kmeans = KMeans(n_clusters=3, random_state=42)
data['Cluster'] = kmeans.fit_predict(X_scaled)


print(data[['QUANTITYORDERED', 'PRICEEACH', 'SALES', 'Cluster']].head())

plt.scatter(X_scaled[:, 0], X_scaled[:, 2], c=data['Cluster'], cmap='rainbow')
plt.xlabel('Quantity Ordered (scaled)')
plt.ylabel('Sales (scaled)')
plt.title('K-Means Clustering on Sales Data')
plt.show()
